Linux
	Enable automatic updates
		sudo apt install unattended-upgrades
		sudo dpkg-reconfigure --priority=low unattended-upgrades
	Create new user/Add user to sudoers
		adduser [USER]
		usermod -aG sudo [USER]
	ssh keys
		mkrdir ~/.ssh && chmod 700 ~/.ssh
		ssh-keygen -b 4096
		ssh-copy-id [USER]@[SERVER]
	Lockdown logins
		sudo vim /etc/ssh/sshd_config	
		- Change port #
		- AddressFamily inet
		- PermitRootLogin no
		- PasswordAuthentication no
		- Protocol2
		- IgnoreRhosts to ues
		- HostbasedAuthentication no
		- PermitEmptyPasswords no
		- X11Forwarding no
		- MaxAuthTries 5
		- Ciphers aes128-ctr,aes192-ctr,aes256-ctr
		- ClientAliveInterval 900
		- ClientAliveCountMax 0
		- UsePAM yes
		chown root:root /etc/ssh/sshd_config
		chmod 600 /etc/ssh/sshd_config
		sudo systemctl restart sshd
		-test ssh (use port number with -p)
	Enable firewall
		sudo ss -tupln
		-check ports/services
		sudo apt install ufw
		sudo ufw allow [PORT]
		sudo ufw enable
		sudo ufw status
		-test ssh
	Block pings 
		sudo vim /etc/ufw/before.rules
		-Add line
			-A ufw-before-input -p icmp --icmp-type echo-request -j DROP
		sudo reboot
	Lock the boot directory
		sudo vim /etc/fstab
		-Add line
			LABEL=/boot	/boot	ext2	defaults,ro	1 2
		sudo chown root:root /etc/grub.conf
		chmod og-rwx /etc/grub.conf
		sed -i "/SINGLE/s/sushell/sulogin" /etc/sysconfig/init
		sed -i "/PROMPT/s/yes/no/" /etc/sysconfig/init
	Disable USB
		vim /etc/modeprobe.d/blacklist.conf
		-Add line 
			blacklist usb_storage
		vim /etc/rc.local
		-Add lines 
			modprobe -r usb_storage
			exit 0
	Check remove services
		apt-cache pkgnames
		-Telnet
		-RSH
		-NIS
		-TFTP
		-TALK
	Enable SELinux
		vim /etc/selinux/config
		SELINUX=enforcing
	Setup permissions
		chown root:root /etc/anacrontab
		chmod og-rwx /etc/anacrontab
		chown root:root /etc/crontab
		chmod og-rwx /etc/crontab
		chown root:root /etc/cron.hourly
		chmod og-rwx /etc/cron.hourly
		chown root:root /etc/cron.daily
		chmod og-rwx /etc/cron.daily
		chown root:root /etc/cron.weekly
		chmod og-rwx /etc/cron.weekly
		chown root:root /etc/cron.monthly
		chmod og-rwx /etc/cron.monthly
		chown root:root /etc/cron.d
		chmod og-rwx /etc/cron.d
